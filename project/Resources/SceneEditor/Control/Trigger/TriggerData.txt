<Trigger>
	<Event>
		<TimeCycle>
			<Number>10</Number>
		</TimeCycle>
	</Event>
	<Condition>
		<And>
			<NumberEqual>
				<Add>
					<GlobalNumber>var</GlobalNumber>
					<Number>3</Number>
				</Add>
				<Number>10</Number>
			</NumberEqual>
			<True/>
		</And>
	</Condition>
	<Action>
		<AssignLocal>
			<LocalNumber>num</LocalNumber>
			<Number>3</Number>
		<AssignLocal>
		<If>
			<Condition>
				<Equal>
					<LocalNumber>num</LocalNumber>
					<Number>3</Number>
				</Equal>
				<True/>
			</Condition>
			<Action>
				<Loop>
					<Number>3</Number>
					<Number>10</Number>
					<Number>1</Number>
					<Action>
						<Sleep>10</Sleep>
						<Print>Hello</Print>
					</Action>
				<Loop>
			</Action>
			<Action>
				<DoNothing/>
			</Action>
		</If>
	</Action>
</Trigger>
